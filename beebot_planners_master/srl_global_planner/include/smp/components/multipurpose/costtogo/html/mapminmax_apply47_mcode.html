61% Map Minimum and Maximum Input Processing Function
62function y = mapminmax_apply(x,settings_gain,settings_xoffset,settings_ymin)
63  y = bsxfun(@minus,x,settings_xoffset);
64  y = bsxfun(@times,y,settings_gain);
65  y = bsxfun(@plus,y,settings_ymin);
66end
